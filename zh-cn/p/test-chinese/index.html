        const generateRandomData = () => {
            const names = ['John', 'Alice', 'Bob', 'Emma', 'David', 'Sarah', 'Michael', 'Lisa'];
            const titles = ['Project Manager', 'Developer', 'Designer', 'Analyst', 'Engineer'];
            const companies = ['Tech Corp', 'Digital Solutions', 'Innovation Labs', 'Future Systems'];
            
            return {
                name: names[Math.floor(Math.random() * names.length)],
                title: titles[Math.floor(Math.random() * titles.length)],
                company: companies[Math.floor(Math.random() * companies.length)],
                score: Math.floor(Math.random() * 100),
                date: new Date(Date.now() - Math.floor(Math.random() * 10000000000)).toLocaleDateString()
            };
        };

        class VirtualList {
            constructor(options) {
                this.options = {
                    container: null,
                    totalItems: 0,
                    itemHeight: 80,
                    visibleItems: 0,
                    buffer: 0,
                    renderItem: () => '',
                    ...options
                }

                this.state = {
                    items: [],
                    startIndex: 0,
                    endIndex: 0,
                    scrollTop: 0,
                    renderTime: 0,
                    totalHeight: 0
                }

                this.init()
            }

            init() {
                const containerHeight = this.options.container.clientHeight
                this.options.visibleItems = Math.ceil(containerHeight / this.options.itemHeight)
                this.options.buffer = Math.ceil(this.options.visibleItems / 2)

                this.state.totalHeight = this.options.totalItems * this.options.itemHeight

                this.createObservers()
                this.attachScrollHandler()
                this.generateInitialData()
                this.render()
                this.updateStats()
            }

            createObservers() {
                const options = {
                    root: this.options.container,
                    threshold: 0
                }

                this.topObserver = new IntersectionObserver((entries) => {
                    entries.forEach(entry => {
                        if (entry.isIntersecting && this.state.startIndex > 0) {
                            this.loadPrevious()
                        }
                    })
                }, options)

                this.bottomObserver = new IntersectionObserver((entries) => {
                    entries.forEach(entry => {
                        if (entry.isIntersecting && this.state.endIndex < this.options.totalItems) {
                            this.loadNext()
                        }
                    })
                }, options)

                this.topObserver.observe(document.getElementById('topSentinel'))
                this.bottomObserver.observe(document.getElementById('bottomSentinel'))
            }

            attachScrollHandler() {
                this.options.container.addEventListener('scroll', () => {
                    requestAnimationFrame(() => {
                        this.handleScroll()
                    })
                })
            }

            handleScroll() {
                const scrollTop = this.options.container.scrollTop
                const newStartIndex = Math.floor(scrollTop / this.options.itemHeight)
                const newEndIndex = Math.min(
                    newStartIndex + this.options.visibleItems + 2 * this.options.buffer,
                    this.options.totalItems
                )

                if (newStartIndex !== this.state.startIndex) {
                    this.state.startIndex = Math.max(0, newStartIndex - this.options.buffer)
                    this.state.endIndex = newEndIndex
                    this.state.items = this.generateItems(this.state.startIndex, this.state.endIndex)
                    this.render()
                    this.updateStats()
                }
            }

            generateInitialData() {
                this.state.startIndex = 0
                this.state.endIndex = Math.min(this.options.visibleItems + 2 * this.options.buffer, this.options.totalItems)
                this.state.items = this.generateItems(this.state.startIndex, this.state.endIndex)
            }

            generateItems(start, end) {
                return Array.from({ length: end - start }, (_, index) => ({
                    id: start + index,
                    data: generateRandomData(),
                }))
            }

            loadPrevious() {
                const newStart = Math.max(0, this.state.startIndex - this.options.buffer)
                const itemsToAdd = this.generateItems(newStart, this.state.startIndex)
                this.state.items = [...itemsToAdd, ...this.state.items.slice(0, this.options.visibleItems)]
                this.state.startIndex = newStart
                this.render()
                this.updateStats()
            }

            loadNext() {
                const newEnd = Math.min(
                    this.options.totalItems,
                    this.state.endIndex + this.options.buffer
                )
                const itemsToAdd = this.generateItems(this.state.endIndex, newEnd)
                this.state.items = [...this.state.items.slice(-this.options.visibleItems), ...itemsToAdd]
                this.state.endIndex = newEnd
                this.render()
                this.updateStats()
            }

            render() {
                const startTime = performance.now()
                const content = document.getElementById('listContent')
                
                content.style.height = `${this.state.totalHeight}px`
                
                const translateY = this.state.startIndex * this.options.itemHeight

                content.innerHTML = `
                    <div style="position: absolute; top: ${translateY}px; width: 100%;">
                        ${this.state.items.map(item => `
                            <div class="list-item" data-id="${item.id}">
                                <div class="list-item-content">
                                    <div class="list-item-title">
                                        ${item.data.name} - ${item.data.title} (${item.id + 1}/10000)
                                    </div>
                                    <div class="list-item-subtitle">${item.data.company}</div>
                                </div>
                                <div class="list-item-meta">
                                    <div>Score: ${item.data.score}</div>
                                    <div>${item.data.date}</div>
                                </div>
                            </div>
                        `).join('')}
                    </div>
                `
                this.state.renderTime = performance.now() - startTime
            }

            updateStats() {
                document.getElementById('totalItems').textContent = this.options.totalItems.toLocaleString()
                document.getElementById('renderedItems').textContent = this.state.items.length.toLocaleString()
                document.getElementById('renderTime').textContent = `${this.state.renderTime.toFixed(2)} ms`
                
                if (window.performance && window.performance.memory) {
                    const memoryUsage = window.performance.memory.usedJSHeapSize / (1024 * 1024)
                    document.getElementById('memoryUsage').textContent = `${memoryUsage.toFixed(2)} MB`
                }
            }

            scrollToIndex(index) {
                const targetIndex = Math.max(0, Math.min(index, this.options.totalItems - 1))
                this.state.startIndex = Math.max(0, targetIndex - this.options.buffer)
                this.state.endIndex = Math.min(
                    this.options.totalItems,
                    targetIndex + this.options.visibleItems + this.options.buffer
                )
                this.state.items = this.generateItems(this.state.startIndex, this.state.endIndex)
                this.render()
                this.updateStats()
                
                const targetElement = document.querySelector(`[data-id="${targetIndex}"]`)
                if (targetElement) {
                    targetElement.scrollIntoView({ behavior: 'smooth', block: 'center' })
                }
            }
        }

        // ÂàõÂª∫ËôöÊãüÂàóË°®ÂÆû‰æã
        const virtualList = new VirtualList({
            container: document.getElementById('virtualList'),
            totalItems: 10000,
            itemHeight: 80,
        })

        // ÁîüÊàêÊñ∞Êï∞ÊçÆ
        function generateNewData() {
            virtualList.generateInitialData()
            virtualList.render()
            virtualList.updateStats()
        }

        // ÊªöÂä®Âà∞ÈöèÊú∫‰ΩçÁΩÆ
        function scrollToRandom() {
            const randomIndex = Math.floor(Math.random() * virtualList.options.totalItems)
            virtualList.scrollToIndex(randomIndex)
        }
